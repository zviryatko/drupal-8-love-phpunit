/** @group simple_block */
class SimpleBlockTest extends BrowserTestBase {
  public static $modules = ['block', 'simple_block'];

  public function testAnonymous() {
    $this->drupalGet('');

    $this->assertSession()->responseContains($link);
  }

    $this->drupalLogin($this->user);
    $this->drupalGet('');
