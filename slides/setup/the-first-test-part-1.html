The Simple Block

class SimpleBlock extends BlockBase {
  public function build() {
    if (\Drupal::currentUser()->isAnonymous()) {
      return (Link::createFromRoute(
        $this->t('Sign in'),
        'user.login')
      )->toRenderable();
    }
    else {
      return [
        '#markup' => $this->t('Hello, %username!', [
          '%username' => \Drupal::currentUser()->getDisplayName(),
        ]),
      ];
    }
  }
}
